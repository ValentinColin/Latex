\section{Coordonnées orthogonales}

\subsection{Coordonnées cartésiennes}
\begin{center}
	\includegraphics{IMAGES/coord-cartesien.png}
\end{center}
\begin{itemize}
	\item base : $\widehat{e_{x}},\widehat{e_{y}},\widehat{e_{z}}$
	\item coordonnées : x,y,z
	\item déplacement élémentaire : \[\vect{\dif l} = \dif x~\widehat{e_{x}} + \dif y~\widehat{e_{y}} + \dif z~\widehat{e_{z}}\]
\end{itemize}

\subsection{Coordonnées cylindriques}
\begin{center}
	\includegraphics{IMAGES/coord-cylindrique.png}
\end{center}
\begin{itemize}
	\item base : $\widehat{e_{\rho}},\widehat{e_{\varphi}},\widehat{e_{z}}$
	\item coordonnées : $\rho,\varphi,z$
	\item déplacement élémentaire : \[\vect{\dif l} = \dif\rho~\widehat{e_{\rho}} + \rho~\dif\varphi~\widehat{e_{\varphi}} + \dif z~\widehat{e_{z}}\]
\end{itemize}

\subsection{Coordonnées sphériques}
\begin{center}
	\includegraphics{IMAGES/coord-spherique.png}
\end{center}
\begin{itemize}
	\item base : $\widehat{e_{r}},\widehat{e_{\theta}},\widehat{e_{\varphi}}$
	\item coordonnées : $r,\theta,\varphi$
	\item déplacement élémentaire : \[\vect{\dif l} = \dif r~\widehat{e_{r}} + r~\dif\theta~\widehat{e_{\theta}} + r\sin\theta~\dif\varphi~\widehat{e_{\varphi}}\]
\end{itemize}

\subsection{Notation générale}
\begin{itemize}
	\item base : $\widehat{e_{1}},\widehat{e_{2}},\widehat{e_{3}}$
	\item coordonnées : $s_{1},s_{2},s_{3}$
	\item déplacement élémentaire : \[\vect{\dif l} = \sum_{i=1}^{3} {\mu_{i}\dif s_{i}\widehat{e_{i}}} \]
\end{itemize}
où $\mu_{i}$ est le \underline{multiplicateur} de la coordonnée $s_{i}$.
\begin{center}
\begin{tabular}{|c|c|c|c|}
\hline
\backslashbox{repérage}{multiplicateurs} & $\mu_{1}$ & $\mu_{2}$ & $\mu_{3}$ \\
\hline
cartésien : $s_{1}$ = x, $s_{2}$ = y, $s_{3}$ = z & 1 & 1 & 1 \\
\hline
cylindrique : $s_{1}$ = $\rho$, $s_{2}$ = $\varphi$, $s_{3}$ = z & 1 & $\rho$ & 1 \\
\hline
sphérique : $s_{1}$ = r, $s_{2}$ = $\theta$, $s_{3}$ = $\varphi$ & 1 & r & r~$\sin\theta$ \\
\hline
\end{tabular}
\end{center}

\section{Théorèmes intégraux}

\subsection{Théorème de Gauss}
\begin{equation*}
\varoiint_{S} {\vect{E}\vect{\dif s}} = \frac{Q_{int}}{\epsilon_{0}}
\end{equation*}

\subsection{Théorème d'Ampère}
\begin{equation*}
\oint_{C} {\vect{B}\vect{\dif l}} = \mu_{0}\sum{I_{enlacés}}
\end{equation*}

\subsection{Théorème de Stokes}
\begin{center}
\[\oint_{C}{\vect{E}\vect{\dif l}} = \iint_{S}{\vect{rot}\left[\vect{E}(M)\right]\vect{\dif s}}\]
\end{center}

\subsection{Théorème de Kelvin}
\begin{center}
Equivalent du théorème de Stokes pour les fonctions scalaires
\[\oint_{C}{f(M)}\vect{\dif l} = \iint_{S}{\vect{\dif s}} \wedge \vect{grad}~f(M)\]
\end{center}

\subsection{Théorème d'Ostrogradski}
\begin{center}
\[\iiint_{V}{div\left[\vect{E}(M)\right]}\dif V = \varoiint_{S}{\vect{E}(M)\vect{\dif s}}\]
\end{center}

\subsection{Théorème du gradiant}
\begin{center}
Equivalent du théorème d'Ostrogradski pour les fonctions scalaires
\[\iiint_{V}{\vect{grad}\left[f(M)\right]}\dif V = \varoiint_{S}{f(M)\vect{\dif s}}\]
\end{center}

\subsection{Théorème du rotationnel}
\begin{center}
\[\varoiint_{S}{\vect{E}(M) \wedge \vect{\dif s}} = -\iiint_{S}{\vect{rot}\left[\vect{E}(M)\right]\dif V}\]
\end{center}

\section{Rappel: double produit vectoriel}
\[\vect{A} \wedge (\vect{B} \wedge \vect{C}) = \vect{B}(\vect{A}\vect{C}) - \vect{C}(\vect{A}\vect{B})\]

	\vect{E}(M) =& \dfrac{1}{4\pi \epsilon_0}\displaystyle\iiint_{\mathcal{D}}{\rho(P) \dfrac{\vect{PM}}{PM^3}}\dif\tau(P)\\
	\vect{B}(M) =& \dfrac{\mu_0}{4\pi}\displaystyle\iiint_{\mathcal{D}}{\dfrac{\vec\jmath~(P) \wedge \vect{PM}}{PM^3}}\dif\tau(P)
\end{align*}

\subsection{Lien entre champ et potentiel}
\begin{center}
Soit $\vect{E}$ (resp. $\vect{B}$) un champ électrique (resp. magnétique) et V (resp. $\vect{A}$) un potentiel électrique \\
(resp. magnétique) associé à $\vect{E}$ (resp. à $\vect{B}$).
\end{center}
\begin{align*}
	\vect{E} =& -\vect{grad}~V - \frac{\partial \vect{A}}{\partial t}\\
	\vect{B} =& ~rot(\vect{A})
\end{align*}


\subsection{Équations de Poisson pour le potentiel scalaire/vecteur}
\begin{align*}
	\Delta V =& -\frac{\rho}{\epsilon_{0}}\\
	\Delta \vect{A}  =& -\mu_{0}\vec{\jmath}
\end{align*}

\subsection{Rappels}
\begin{itemize}
	\item $div\vect{B} = 0$ (car $\vect{B} = \vect{rot}\vect{A}$)\\[3mm]
	\item $\vect{rot}\vect{B} = \mu_0 \vec{\jmath}$ \\[1mm]
	\item Biot et Savart : $\vect{B}(M) = \dfrac{\mu_0}{4\pi}\displaystyle\iiint_{\mathcal{D}}{\dfrac{\vec\jmath~(P) \wedge \vect{PM}}{PM^3}}\dif\tau(P)$
\end{itemize}

\subsection{Force de Lorentz}
\begin{equation*}
\vect{F} = q\vect{E} + q\vect{v} \wedge \vect{B}
\end{equation*}

\subsection{Force de Laplace}
La force de Laplace est la résultante de toutes les forces de Lorentz sur les charges mobiles.
\begin{equation*}
\vect{\dif F} = I\vect{\dif l} \wedge \vect{B}
\end{equation*}
Cette force agit directement sur les charges fixes du matériaux (ex. protons)

\subsection{Loi de Lenz}
\begin{law}[Loi de Lenz]
	Les conséquences de l'induction s'opposent aux causes qui lui donne naissance.
\end{law}

\subsection{Force électromotrice}
\begin{law}[Loi de Faraday]
	Soit $\phi$ un flux magnétique alors la force électromotrice (induite) e est égale à l'opposer de la variation temporelle du flux magnétique.
	\[e = -\frac{\dif \phi}{\dif t}\]
	(Cette loi n'est pas toujours valide notamment lors de circuit à contact glissant)
\end{law}
On peut aussi écrire cette f.e.m (écriture toujours valide) :
\begin{align*}
	e &= \int_{Circuit}{\hspace{-6.5mm}\vect{E_m}~\vect{\dif l}}\\
	\text{tel que}\quad \vect{E_m} &= \vect{v} \wedge \vect{B} - \frac{\partial \vect{A}}{\partial t}
\end{align*}
Où $\vect{E_m}$ est le champ \underline{électromoteur} (induit) le long du circuit


\subsection{Potentiels et choix de jauge}
\subsubsection{Choix de jauge}

\begin{proposition}[Jauge de Lorentz]
Soit V et $\vect{A}$ un potentiel scalaire et vecteur. On impose donc :
\[div \vect{A} + \dfrac{1}{c^2}\dfrac{\partial V}{\partial t} = 0\]
\end{proposition}

\begin{proposition}[Jauge de Coulomb]
Soit $\vect{A}$ un potentiel vecteur. On impose donc :
\[div \vect{A} = 0\]
\end{proposition}

\subsubsection{D'Alembertiens des Potentiels}
En Jauge de Lorentz en retrouve les équation de Poisson.
\begin{align*}
	\square V &= -\dfrac{\rho}{\epsilon_0}\\
	\square \vect{A} &= -\mu_0 \vec{\jmath}
\end{align*}

\subsubsection{D'Alembertiens des Champs Électriques et Magnétiques}
Équation de D'Alembert du champ Électrique et Magnétique.
\begin{align*}
	\square \vect{E} &= \vect{0}\\
	\square \vect{B} &= \vect{0}
\end{align*}
